Some examples :

"0" = > true

" 0.1 " = > true

"abc" = > false

"1 a" = > false

"2e10" = > true

bool isDigit(char ch)
{
	if (ch >= '0' && ch <= '9')
	{
		return true;
	}
	return false;
}

bool isN(string s, int &n)
{
	if (s.size() == 0)
	{
		return true;
	}

	if (isDigit(s[0]))
	{
		n++;
		return isN(s.substr(1), n);
	}

	if (s[0] == '.' || s[0] == ' ')
		return isN(s.substr(1), n);
	if (s[0] == 'e')
	{
		if (s.size() == 1)
		{
			return false;
		}
		else if ((s[1] == '+' || s[1] == '-') && s.size() == 2)
			return false;
		else if ((s[1] == '+' || s[1] == '-') && s.size() > 2)
		{
			return isN(s.substr(2), n);
		}
		else
		{
			return isN(s.substr(1), n);
		}
		
	}

	return false;
}

bool isNumber(string s) 
{
	int i = 0;
	while (s[i] == ' ')
	{
		i++;
	}

	if (i==s.size())
	{
		return false;
	}

	if (countE > 1 || countPoint > 1)
		return false;
	
	int n = 0;
	if (s[i] == '+' || s[i] == '-')
	{
		if (s[i+1] == 'e')
		{
			return false;
		}

		return isN(s.substr(i+1), n) && n!=0;
	}

	if (s[i] == 'e')
	{
		return false;
	}

	if (s[i] == '.')
	{
		return isN(s.substr(i + 1), n) && n != 0;
	}

	return isN(s, n);
}

{
	int pos = 0;

	while (s[pos] == ' ')
	{
		pos++;
	}

	if (s[pos] == '+' || s[pos] == '-')
	{
		pos++;
	}

	int countN = 0;

	{
		if (isdigit(s[pos]))
		{
			countN++;
		}
		if (s[pos] == '.')
		{
			countP++;
		}

		pos++;
	}

	if (!countN || countP>1)
	{
		return false;
	}

		}
	}

		pos++;
	}
